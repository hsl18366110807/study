## for循环

```c++
for ([[maybe_unused]] int n : a)
    std::cout << 1 << ' '; // the loop variable need not be used
std::cout << '\n';

for (auto n = v.size(); auto i : v)
{
    // the init-statement (C++20)
    std::cout << --n + i << ' ';
}
for (auto&& [first, second] : mymap)
{
    // use first and second
}
```
